23// Optional: allow redirecting output to a file at runtime when enabled
24static inline FILE* logger_output()
25{
26 static FILE* out = NULL;
27 if (!out) out = stderr; // default to stderr
28 return out;
29}

31static inline void logger_set_output_file(FILE* f)

37static inline void logger_vlog(const char* comp, const char* level, const char* fmt, va_list ap)
38{
39 // Timestamp in microseconds
40 uint64_t ts_us = get_timestamp_ns() / 1000;
41 FILE* out = logger_output();
42 flockfile(out);
43 fprintf(out, "[%llu.%06llu] [%s] [%s] ",
44 (unsigned long long)(ts_us / 1000000ULL),
45 (unsigned long long)(ts_us % 1000000ULL),
46 level, comp);
47 vfprintf(out, fmt, ap);
48 fputc('\n', out);
49 funlockfile(out);
50}

52static inline void logger_log(const char* comp, const char* level, const char* fmt, ...)
53{
54 va_list ap; va_start(ap, fmt);
55 logger_vlog(comp, level, fmt, ap);
56 va_end(ap);
57}
